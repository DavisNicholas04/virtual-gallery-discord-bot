### JSON file format:
````json
{
    "history": [
        {
            "title":"string",
            "description":"string",
	    "date":"dateTime",
            "images":"string <location of file in github>"
	}
    ],

    "anime": [
        {
            "title":"string",
            "description":"string",
            "release_date":"dateTime",
            "images":"string <location of file in github>",
            "genre":"string"
        }
    ],

    "game": [
        {
            "title":"string",
            "description":"string",
            "release_date":"dateTime",
            "images":"string <location of file in github>",
            "genre":"string"
        }
    ]
}
````
